node-youtube-data

Get data from YouTube as JSON

Examples

yt = require('youtube-data')

yt.query()
  .videos('monstercatmedia')
  .results(50)
  .orderByPublished()
  .simple()
  .run (err, data) ->
    return console.log err if err

    console.log data.feed.entry
